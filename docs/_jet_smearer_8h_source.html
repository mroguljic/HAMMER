107 RVec<float> JetSmearer::GetSmearValsPt(LorentzV jet, LorentzV genJet, float rho) {
108  RVec<float> out = {};
109  if (jet.Pt() <= 0.){
110  printf("WARNING: jet pT = %f !!", jet.Pt());
111  out = {1.,1.,1.};
112  }
113 
114  int central_or_shift;
115  std::vector<int> variation_index = {0,2,1}; // nom,up,down
116 
117  std::map<int, float> jet_pt_sf_and_uncertainty = {};
118  for (size_t i = 0; i < 3; i++) {
119  central_or_shift = variation_index[i];
120  params_sf_and_uncertainty.setJetEta(jet.Eta());
121  params_sf_and_uncertainty.setJetPt(jet.Pt());
122 
123  jet_pt_sf_and_uncertainty[central_or_shift] =
124  jerSF.getScaleFactor(params_sf_and_uncertainty, central_or_shift);
125  }
126 
127  std::map<int, float> smear_vals = {};
128  float jet_pt_resolution, smear_factor;
129  if (genJet) { // Case 1
130  for (size_t i = 0; i < 3; i++) {
131  central_or_shift = variation_index[i];
132  float dPt = jet.Pt() - genJet.Pt();
133  smear_factor = 1. + (jet_pt_sf_and_uncertainty[central_or_shift] - 1.) * dPt / jet.Pt();
134  smear_vals[central_or_shift] = smear_factor;
135  }
136  } else {
137  params_resolution.setJetPt(jet.Pt());
138  params_resolution.setJetEta(jet.Eta());
139  params_resolution.setRho(rho);
140  jet_pt_resolution = jer.getResolution(params_resolution);
141 
142  float rand = rnd.Gaus(0, jet_pt_resolution);
143 
144  for (size_t i = 0; i < 3; i++) {
145  central_or_shift = variation_index[i];
146 
147  if (jet_pt_sf_and_uncertainty[central_or_shift] > 1.){ // Case 2
148  smear_factor =
149  1. + rand * sqrt( pow(jet_pt_sf_and_uncertainty[central_or_shift],2) - 1.);
150  } else { // Case 3
151  smear_factor = 1.;
152  }
153  smear_vals[central_or_shift] = smear_factor;
154  }
155  }
156 
157  // check that smeared jet energy remains positive, as the direction of
158  // the jet would change ("flip") otherwise - and this is not what we want
159  for (size_t i = 0; i < 3; i++) {
160  central_or_shift = variation_index[i];
161  if (smear_vals[central_or_shift] * jet.E() < 1.e-2){
162  smear_vals[central_or_shift] = 1.e-2 / jet.E();
163  }
164  out.push_back(smear_vals[central_or_shift]);
165  }
166  return out;
167 }

230  }
231  }
