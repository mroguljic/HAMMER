166 void GenParticleTree::AddParticle(Particle* particle) {
167  Particle* staged_node = particle;
168 
169  // First check if current heads don't have new parents and
170  // remove them from heads if they do
171  std::vector<int> heads_to_delete {};
172  for (int i = 0; i < heads.size(); i++) {
173  if (heads.at(i)->parentIndex == staged_node->index) {
174  heads_to_delete.push_back(i);
175  }
176  }
177  // Need to delete indices in reverse order so as not to shift
178  // indices after a deletion
179  for (int ih = heads_to_delete.size(); ih >= 0; ih--) {
180  heads.erase(heads.begin()+ih);
181  }
182 
183  // Next identify staged node has no parent (heads)
184  // If no parent, no other infor we can get from this particle
185  std::vector<int> indexes = StoredIndexes();
186  if (InList(staged_node->parentIndex, indexes)) {
187  heads.push_back(staged_node);
188  nodes.push_back(staged_node);
189  } else {
190  for (int inode = 0; inode < nodes.size(); inode++) {
191  if (staged_node->parentIndex == nodes[inode]->index){
192  staged_node->AddParent(inode);
193  nodes.at(inode)->AddChild(nodes.size());
194  nodes.push_back(staged_node);
195  }
196  }
